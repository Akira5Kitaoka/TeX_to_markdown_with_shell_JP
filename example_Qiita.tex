\begin{theorem}
    \label{theo:test}
    二次方程式
    \[
        x^2 + 2 b x +c = 0 
    \]
    の解は
    \[
        x = - b \pm \sqrt{b^2-c }
        .
    \]
\end{theorem}

\begin{proof}
    \begin{align*}
        & 
        x^2 + 2 b x +c = 0 
        \\
        & \Leftrightarrow
        (x + b)^2 -b^2 +c = 0 
        \\
        & \Leftrightarrow
        (x + b)^2 = b^2 -c
        \\
        & \Leftrightarrow
        x + b = \pm \sqrt{b^2 -c} %ルートをとる．
        \\
        & \Leftrightarrow
        x = -b \pm \sqrt{b^2 -c}
        .  
    \end{align*}
\end{proof}
